FROM ubuntu:20.04		# указывает image с которого берётся сборка.
USER admilya:admin		# от какого пользователя и группы запускаются команды. по умолчанию root.
EXPOSE 8080/tcp			# информирует движок о том, какой порт image планируется открыть наружу и протокол. Порт не открывается, он только определяется. Порт наружу открывается через -p  при запуске контейнера.
WORKDIR application		# директория из которой выполнять инструкции. Относительный путь будет применяться относительно предыдущей инструкции. Аналог cd.
ARG PROFILE=docker		# переменная, которую можно передать при запуске docker build --biuld-arg<var name>=<value>. 
ADD https://github.com/*/name	# чтобы скопировать внутрь контейнера. Может со внешних источников, умееь разархивировать файлы.
COPY build/*/*.jar		# чтобы скопировать внутрь контейнера с Host_Machine.
ENV S_P_ACTIVE=$PROFILE		# Переменная окружения для запущенного контейнера.
RUN apt-get update && apt-***	# выполнит команду в новом слое, поверх текущего. Каждый RUN это слой образа *.
docker run <name of docker>
docker run <name of docker> ya.ru
ENTRYPOINT ["/bin/ping"]	# описывает контейнер, как исполняемый. Если нет дополнительных параметров запуска, то используется CMD, как defalt arg. Если есть парамтры запуска, то они приоритетней, чем аргументы внутри.
CMD ["localhost"]		# дефолтная команда при старте.
